• 利用API: Yahoo! JavaScriptマップAPI
  • ガイド: Yahoo! JavaScriptマップAPIでOpenStreetMapを使う
window.onload = function() {
    // OpenStreetMapを有効にするためのオプションを指定します
    var map = new Y.Map("map",{
        "configure": {
            "enableOpenStreetMap": true,
            "scrollWheelZoom": true
        }
    });

    // 地図の初期表示にOpenStreetMapを指定して描画します
    map.drawMap(new Y.LatLng(35.680840,139.767009), 14, Y.LayerSetId.OSM);
    map.addControl(new Y.LayerSetControl());
};
